        let currentUnit = 3; let score = 0; let completed = new Set();

        function loadUnit(unitId) {
            currentUnit = unitId; score = 0; completed.clear();
            document.querySelectorAll('.nav-btn').forEach(btn => {
                btn.classList.remove('active', 'bg-indigo-600', 'text-white');
                btn.classList.add('bg-white', 'text-slate-600');
            });
            document.getElementById(`nav-${unitId}`).classList.add('active');
            renderTasks(); updateStats();
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function renderTasks() {
            const container = document.getElementById('task-container');
            container.innerHTML = `<h2 class="text-xl font-bold text-slate-800 mb-6 px-2 border-l-4 border-indigo-500 pl-4">${database[currentUnit].title}</h2>`;
            database[currentUnit].questions.forEach((item, idx) => {
                const card = document.createElement('div');
                card.className = "custom-card p-6 rounded-[2rem] mb-6 shadow-sm";
                let html = `<p class="text-lg font-bold text-slate-800 mb-5 leading-relaxed">${item.q}</p>`;
                
                if(item.type === 'fill') {
                    html += `<div class="flex flex-col sm:flex-row gap-3">
                        <input type="text" id="ans-${idx}" placeholder="Type answer..." class="flex-1 bg-slate-50 border-2 border-transparent rounded-2xl px-5 py-3 outline-none focus:border-indigo-400 transition-all font-medium">
                        <button onclick="checkFill(${idx}, '${item.a}')" class="px-8 py-3 bg-slate-900 text-white rounded-2xl font-bold hover:bg-indigo-600 transition-all active:scale-95">Check</button>
                    </div><p id="fdb-${idx}" class="mt-3 text-sm font-bold hidden"></p>`;
                } else {
                    html += `<div class="grid grid-cols-1 gap-3">`;
                    item.opts.forEach((opt, oi) => {
                        html += `<button onclick="checkMCQ(${idx}, ${oi}, ${item.a}, this)" class="option-btn w-full text-left p-4 rounded-2xl bg-white flex items-center group">
                            <span class="w-8 h-8 rounded-lg bg-slate-100 flex items-center justify-center mr-4 text-xs font-black text-slate-500 group-hover:bg-indigo-500 group-hover:text-white transition-all">${String.fromCharCode(65+oi)}</span>
                            <span class="font-semibold text-slate-700">${opt}</span>
                        </button>`;
                    });
                    html += `</div>`;
                }
                card.innerHTML = html; container.appendChild(card);
            });
        }

        function checkFill(idx, cor) {
            const key = `u${currentUnit}-q${idx}`;
            if(completed.has(key)) return;
            const input = document.getElementById(`ans-${idx}`);
            const fdb = document.getElementById(`fdb-${idx}`);
            if(input.value.trim().toLowerCase() === cor.toLowerCase()) {
                input.className = "flex-1 bg-green-50 border-2 border-green-500 rounded-2xl px-5 py-3 text-green-700 font-bold";
                score += 10; fdb.innerText = "Correct! ✨"; fdb.className = "mt-3 text-sm font-bold text-green-600";
            } else {
                input.className = "flex-1 bg-red-50 border-2 border-red-500 rounded-2xl px-5 py-3 text-red-700 font-bold";
                fdb.innerText = `Incorrect. Ans: ${cor}`; fdb.className = "mt-3 text-sm font-bold text-red-600";
            }
            fdb.classList.remove('hidden'); completed.add(key); updateStats();
        }

        function checkMCQ(qIdx, sel, cor, btn) {
            const key = `u${currentUnit}-q${qIdx}`;
            if(completed.has(key)) return;
            const btns = btn.parentElement.querySelectorAll('button');
            if(sel === cor) { btn.classList.add('correct'); score += 10; } 
            else { btn.classList.add('wrong'); btns[cor].classList.add('correct'); }
            btns.forEach(b => b.disabled = true); completed.add(key); updateStats();
        }

        function updateStats() {
            document.getElementById('score-val').innerText = score;
            const total = database[currentUnit].questions.length;
            document.getElementById('progress-val').innerText = `${completed.size}/${total}`;
        }

        window.onload = () => loadUnit(3);
